30  void insert(const Entity entity, const T component) {
31  assert(entityToIndexMap.find(entity) == entityToIndexMap.end() &&
32  "Component added to same entity more than once.");
33 
34  size_t newIndex = totalSize;
35  entityToIndexMap[entity] = newIndex;
36  indexToEntityMap[newIndex].insert(entity);
37  componentArray[newIndex] = component;
38  ++totalSize;
39  }
40 
41  void remove(const Entity entity) {
42  assert(entityToIndexMap.find(entity) != entityToIndexMap.end() &&
43  "Removing non-existent component.");
44 
45  size_t indexToRemove = entityToIndexMap[entity];
46  size_t indexOfLastElement = totalSize - 1;
47 
48  if (indexToEntityMap[indexToRemove].size() == 1) {
49  auto &component = componentArray[indexToRemove];
50  componentArray[indexToRemove] = componentArray[indexOfLastElement];
51  auto lastElements = indexToEntityMap[indexOfLastElement];
52  for (auto &en: lastElements) {
53  entityToIndexMap[en] = indexToRemove;
54  }
55  indexToEntityMap[indexToRemove] = lastElements;
56  deleteFunction(component);
57  --totalSize;
58  } else {
59  indexToEntityMap[indexToRemove].erase(entity);
60  }
61  }
62 
63  void copy(const Entity src, const Entity dst) {
64  assert(entityToIndexMap.find(src) != entityToIndexMap.end() &&
65  "Copying non-existent component.");
66  assert(entityToIndexMap.find(dst) == entityToIndexMap.end() &&
67  "Copying to an already existing component.");
68 
69  size_t indexToCopy = entityToIndexMap[src];
70  entityToIndexMap[dst] = indexToCopy;
71  indexToEntityMap[indexToCopy].insert(dst);
72  }
73 
74  T &getComponent(const Entity entity) {
75  assert(entityToIndexMap.find(entity) != entityToIndexMap.end() &&
76  "Retrieving non-existent component.");
77 
78  return componentArray[entityToIndexMap[entity]];
79  }
80 
81  void safeRemove(const Entity &entity) noexcept final {
82  if (entityToIndexMap.find(entity) != entityToIndexMap.end())
83  remove(entity);
84  }
85 
86 private:
87  std::array<T, MAX_ENTITIES> componentArray;
88  std::unordered_map<Entity, size_t> entityToIndexMap;
89  std::unordered_map<size_t, std::set<Entity>> indexToEntityMap;
90  std::function<void(T &)> deleteFunction;
91  size_t totalSize = 0;
92 };
